Marker über JSON-File setzen

Laden einer JSON-Datei mit Geo-Informationen zum erstellen von Markern.

Nur das Beispiel anzeigen
Normalerweise würde man einen JSON File von einer API oder einer anderen externen Quelle bzw. Datenbank laden, in diesem Falle werden die JSON Daten direkt ins Script eingefügt. Zusätzlich zu den üblichen Dateien benötigen wir für dieses Beispiel noch die jQuery-Library.
index.html 
<!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title>Leaflet Basic Setup</title>
		<link rel="stylesheet" type="text/css" href="leaflet/dist/leaflet.css">
		<script type="text/javascript" src="leaflet/dist/leaflet.js"></script>
		<style>
			html, body { margin:0; padding: 0; }
			#map { min-height: 400px; }
		</style>
	</head>
	<body>
		<div id="map"></div>
		<script type="text/javascript" src="map.js"></script>
	</body>
</html>
map.js 
var map = L.map('map', {
	center : [52.52, 13.4],
	zoom : 13
});

L.tileLayer('http://{s}.tile.cloudmade.com/[API-Key]/997/256/{z}/{x}/{y}.png').addTo(map);

var imageUrl = '../img/Napoleons_March.png',
    imageBounds = [[48, 2], [55, 37]];

L.imageOverlay(imageUrl, imageBounds).addTo(map);
